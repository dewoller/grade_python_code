    """
    text = rich.text.Text()

# C1: 2 marks (1 for def and function name, 1 for parameter)
def bot_count(payload):
    # C7: 2 marks for handling a single input
    if len(payload) == 1:
        from_number = 1
        stop = int(payload[0]) + from_number        # C2: 1 mark for integer conversion
    else:
        from_number = int(payload[0])
        stop = int(payload[1]) + from_number        # C2: 1 mark for integer conversion

    # C6: 1 mark for initialising variable for summation
    total_sum = 0

    # C3: 1 mark for using loop
    # C4: 2 marks if loop start/end correct
    for i in range(from_number, stop):
        # C3: 1 mark for bot_say.
        # C5: 2 marks for correct value printed
        bot_say(i)

        # C6: 1 mark for aggregation
        total_sum += i

    # C6: 1 mark for bot_say with the total
    bot_say(f'Sum: {total_sum}')
